CHAOS_PP_BUFFER

The CHAOS_PP_BUFFER macro is used to maintain a buffer of valid recursion steps.

Usage 

CHAOS_PP_BUFFER(size)
size

The size of the buffer to be maintained.

Remarks

This macro is used to produce a recursion state size steps ahead of s.

This recursion state is used by some algorithms to guarantee that user-defined macros (or Î» expressions) have at least size recursion steps available when they are invoked. Such an algorithm increases this recursion state as the actual recursion state increases. When this auxiliary recursion state reaches CHAOS_PP_LIMIT_EXPR, the algorithm performs a parametric or exponential jump.

This macro deduces the current recursion state and invokes CHAOS_PP_BUFFER_S. If the current recursion state is already known, direct use of CHAOS_PP_BUFFER_S is more efficient.

This macro may not be used if bypass semantics are in effect.

#include <chaos/preprocessor/tuple/eat.h>

#define REPEAT(size, count, macro, data) \
    REPEAT_S(CHAOS_PP_STATE(), size, count, macro, data) \
    /**/
#define REPEAT_S(s, size, count, macro, data) \
    REPEAT_I( \
        CHAOS_PP_NEXT(s), CHAOS_PP_BUFFER_S(s, size), \
        count, macro, data \
    ) \
    /**/
#define REPEAT_I(s, buffer, count, macro, data) \
    REPEAT_II( \
        s, s, buffer, buffer, \
        count, macro, CHAOS_PP_CALL(macro), data \
    ) \
    /**/
#define REPEAT_INDIRECT() REPEAT_II
#define REPEAT_II(s, o, buffer, orig, count, macro, _m, data) \
    CHAOS_PP_IIF( \
        CHAOS_PP_NOT_EQUAL(buffer, CHAOS_PP_LIMIT_EXPR) \
    )( \
        CHAOS_PP_IF(count)( \
            REPEAT_A, CHAOS_PP_TUPLE_EAT(9) \
        ), \
        REPEAT_B \
    )( \
        CHAOS_PP_OBSTRUCT(), s, o, buffer, orig, \
        count, macro, _m, data \
    ) \
    /**/
#define REPEAT_A(_, s, o, buffer, orig, count, macro, _m, data) \
    CHAOS_PP_EXPR_S(s) _(REPEAT_INDIRECT _()( \
        CHAOS_PP_NEXT(s), o, CHAOS_PP_NEXT(buffer), orig, \
        CHAOS_PP_DEC(count), macro, _m, data \
    )) \
    _m()(s, macro, CHAOS_PP_DEC(count) CHAOS_PP_COMMA() data) \
    /**/
#define REPEAT_B(_, s, o, buffer, orig, count, macro, _m, data) \
    CHAOS_PP_EXPR_S CHAOS_PP_OBSTRUCT()(o)( \
        REPEAT_INDIRECT CHAOS_PP_OBSTRUCT()()( \
            CHAOS_PP_NEXT(o), o, CHAOS_PP_NEXT(orig), orig, \
            count, macro, _m, data \
        ) \
    ) \
    /**/

#define A(s, n, _) < n >
#define B(s) \
    CHAOS_PP_EXPR_S(s)(CHAOS_PP_EXPR_S(s)( \
        CHAOS_PP_EXPR_S(s)(CHAOS_PP_EXPR_S(s)( \
            REPEAT_S(s, 5, 15, A, ~) \
        )) \
    )) \
    /**/

B(CHAOS_PP_SUB(CHAOS_PP_LIMIT_EXPR, 10))
